local function spawnDeerGod()
    local spawner = loadstring(game:HttpGet("https://raw.githubusercontent.com/RegularVynixu/Utilities/main/Doors/Entity%20Spawner/V2/Source.lua"))()

    local entity = spawner.Create({
        Entity = {
            Name = "Deer God",
            Asset = "rbxassetid://12262883448",
            HeightOffset = 0
        },
        Movement = {
            Speed = 20,
            Delay = 2,
            Reversed = false
        },
        Rebounding = {
            Enabled = false -- Тільки в один бік
        },
        Damage = {
            Enabled = true,
            Range = 50,
            Amount = 100
        },
        CameraShake = {
            Enabled = true,
            Range = 80,
            Values = {2, 20, 0.1, 1}
        },
        Lights = {
            Flicker = { Enabled = true, Duration = 4 },
            Shatter = true, -- Ламає лампи
            Repair = false
        }
    })

    -- Підказка при смерті
    entity:SetCallback("OnKillingPlayer", function()
        pcall(function()
            firesignal(game.ReplicatedStorage.RemotesFolder.DeathHint.OnClientEvent, {"You died to the Deer God...", "He moves slowly. You must hide in time!"}, "Blue")
        end)
    end)

    entity:Run()
end

spawnDeerGod()
