local function spawnRebound()
    local spawner = loadstring(game:HttpGet("https://raw.githubusercontent.com/RegularVynixu/Utilities/main/Doors/Entity%20Spawner/V2/Source.lua"))()

    -- 1. СТВОРЕННЯ СУТНОСТІ REBOUND
    local entity = spawner.Create({
        Entity = {
            Name = "Rebound",
            Asset = "rbxassetid://12254145022", -- Твій ID
            HeightOffset = 0
        },
        Movement = {
            Speed = 100, -- Швидкий, але не як А-60
            Delay = 2,
            Reversed = false
        },
        -- МЕХАНІКА ПОВЕРНЕНЬ (REBOUNDING)
        Rebounding = {
            Enabled = true,
            Type = "Ambush", -- Тип Ambush змушує його літати туди-сюди
            Min = 2, -- Мінімум 2 повернення
            Max = 4, -- Максимум 4 повернення
            Delay = 1
        },
        Damage = {
            Enabled = true,
            Range = 45,
            Amount = 100
        },
        CameraShake = {
            Enabled = true,
            Range = 70,
            Values = {2, 15, 0.1, 1}
        },
        Lights = {
            Flicker = { Enabled = true, Duration = 2 },
            Shatter = true,
            Repair = false
        }
    })

    -- 2. СИНЯ ПІДКАЗКА (РОСІЙСЬКОЮ)
    entity:SetCallback("OnKillingPlayer", function()
        pcall(function()
            firesignal(game.ReplicatedStorage.RemotesFolder.DeathHint.OnClientEvent, {
                "Ты погиб от Rebound...", 
                "Он возвращается несколько раз! Не выходи из шкафа слишком рано."
            }, "Blue")
        end)
    end)

    entity:Run()
end
