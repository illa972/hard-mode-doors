local function spawnCease()
    local TweenService = game:GetService("TweenService")
    local spawner = loadstring(game:HttpGet("https://raw.githubusercontent.com/RegularVynixu/Utilities/main/Doors/Entity%20Spawner/V2/Source.lua"))()

    local latestRoom = game.ReplicatedStorage.GameData.LatestRoom.Value
    for i = 0, latestRoom do
        local room = workspace.CurrentRooms:FindFirstChild(tostring(i))
        if room then
            for _, obj in pairs(room:GetDescendants()) do
                if obj:IsA("Light") then
                    TweenService:Create(obj, TweenInfo.new(3), {Color = Color3.fromRGB(0, 100, 255), Brightness = obj.Brightness * 1.5}):Play()
                elseif obj:A("Part") and (obj.Name == "Neon" or obj.Material == Enum.Material.Neon) then
                    TweenService:Create(obj, TweenInfo.new(3), {Color = Color3.fromRGB(0, 100, 255)}):Play()
                end
            end
        end
    end

    local entity = spawner.Create({
        Entity = { Name = "Cease", Asset = "rbxassetid://12262854624", HeightOffset = 0 },
        Movement = { Speed = 45, Delay = 3, Reversed = false },
        Damage = { Enabled = false },
        Lights = { Flicker = { Enabled = true, Duration = 2 }, Shatter = false }
    })

    entity:SetCallback("OnSpawned", function()
        local p = game.Players.LocalPlayer
        local char = p.Character
        local root = char:WaitForChild("HumanoidRootPart")
        local hum = char:WaitForChild("Humanoid")
        local lastPos = root.Position

        task.spawn(function()
            while workspace:FindFirstChild("Cease") and hum.Health > 0 do
                task.wait(0.1)
                local model = workspace:FindFirstChild("Cease")
                if model and model.PrimaryPart then
                    if (root.Position - model.PrimaryPart.Position).Magnitude < 35 then
                        if (root.Position - lastPos).Magnitude > 0.1 then
                            hum.Health = 0
                            firesignal(game.ReplicatedStorage.RemotesFolder.DeathHint.OnClientEvent, {"You died to Cease...", "You moved! Stay perfectly still when he is near."}, "Blue")
                            break
                        end
                    end
                end
                lastPos = root.Position
            end
        end)
    end)

    entity:Run()
end

spawnCease()
