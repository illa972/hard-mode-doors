local function spawnSilence()
    local spawner = loadstring(game:HttpGet("https://raw.githubusercontent.com/RegularVynixu/Utilities/main/Doors/Entity%20Spawner/V2/Source.lua"))()

    local entity = spawner.Create({
        Entity = {
            Name = "Silence",
            Asset = "rbxassetid://11535812658",
            HeightOffset = 0
        },
        Movement = {
            Speed = 65,
            Delay = 1,
            Reversed = false
        },
        Rebounding = {
            Enabled = true,
            Type = "Ambush", 
            Min = 1, -- 1 повернення (загалом 2 прольоти)
            Max = 1, 
            Delay = 2
        },
        Damage = {
            Enabled = true,
            Range = 40,
            Amount = 100
        },
        CameraShake = {
            Enabled = true,
            Range = 60,
            Values = {2, 15, 0.1, 1}
        },
        Lights = {
            Flicker = { Enabled = true, Duration = 2.5 },
            Shatter = true, -- Ламає світло
            Repair = false
        }
    })

    -- Підказка при смерті
    entity:SetCallback("OnKillingPlayer", function()
        pcall(function()
            firesignal(game.ReplicatedStorage.RemotesFolder.DeathHint.OnClientEvent, {"You died to Silence...", "It came back... Stay hidden until it's gone!"}, "Blue")
        end)
    end)

    -- Ачивка при появі
    pcall(function()
        loadstring(game:HttpGet("https://raw.githubusercontent.com/RegularVynixu/Utilities/main/Doors/Custom%20Achievements/Source.lua"))()({
            Title = "Deadly Silence",
            Desc = "Shhh...",
            Reason = "Encounter Silence.",
            Image = "rbxassetid://11380121147" 
        })
    end)

    entity:Run()
end

spawnSilence()
